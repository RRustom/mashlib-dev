import * as UI from '../../lib'

import { Canvas, Meta, Story } from '@storybook/addon-docs/blocks'
import { rawJsonDecorator } from './decorators'

<Meta title="Media Types" />

## isAudio

[API Documentation](https://solid.github.io/solid-ui/Documentation/api/modules/_widgets_buttons_.html#isaudio)

<Canvas withSource="open">
  <Story name="isAudio" decorators={[rawJsonDecorator]}>
    { () => {
      const type = UI.rdf.namedNode('http://www.w3.org/ns/iana/media-types/audio/')
      UI.store.add(UI.rdf.namedNode('http://example.com/audio.mpeg'), UI.ns.rdf('type'), type)
      return UI.widgets.isAudio(UI.rdf.namedNode('http://example.com/audio.mpeg'))
    }
    }
  </Story>
</Canvas>
